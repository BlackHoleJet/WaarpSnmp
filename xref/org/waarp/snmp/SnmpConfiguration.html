228     private static boolean loadConfig() {
229         XmlValue value = hashConfig.get(SNMP_LOCAL_ADDRESS);
230         @SuppressWarnings("unchecked")
231         List<String> values = (List<String>) value.getList();
232         int length = values.size();
233         if (length == 0) {
234             address = new String[] {
235                 DEFAULTADDRESS };
236         } else {
237             address = new String[length];
238             int nb = 0;
239             address = values.toArray(address);
240             String[] tmp = new String[length];
241             for (int j = 0; j < length; j ++) {
242                 if (address[j] != null && (!address[j].trim().isEmpty())) {
243                     tmp[nb] = address[j];
244                     nb ++;
245                 }
246             }
247             if (nb == 0) {
248                 address = new String[] {
249                     DEFAULTADDRESS };
250             } else if (nb < length) {
251                 // less addresses than intended
252                 address = new String[nb];
253                 System.arraycopy(tmp, 0, address, 0, nb);
254                 /*for (int i = 0; i < nb; i ++) {
255                     address[i] = tmp[i];
256                 }*/
257             }
258         }
259         value = hashConfig.get(SNMP_NBTHREAD);
260         if (value != null && (!value.isEmpty())) {
261             nbThread = value.getInteger();
262             if (nbThread <= 0) nbThread = 4;
263         }
264         value = hashConfig.get(SNMP_FILTERED);
265         if (value != null && (!value.isEmpty())) {
266             isFilterAccessEnabled = value.getBoolean();
267         }
268         value = hashConfig.get(SNMP_USETRAP);
269         if (value != null && (!value.isEmpty())) {
270             isUsingTrap = value.getBoolean();
271         }
272         value = hashConfig.get(SNMP_TRAPLEVEL);
273         if (value != null && (!value.isEmpty())) {
274             trapLevel = value.getInteger();
275         }
276         return true;
277     }

315     /**
316      * 
317      new XmlDecl(XmlType.STRING, SNMP_SECURITY_NAME), new
318      * XmlDecl(XmlType.STRING, SNMP_SECURITY_AUTH_PROTOCOL), new
319      * XmlDecl(XmlType.STRING, SNMP_SECURITY_AUTH_PASSPHRASE), new
320      * XmlDecl(XmlType.STRING, SNMP_SECURITY_PRIV_PROTOCOL), new
321      * XmlDecl(XmlType.STRING, SNMP_SECURITY_PRIV_PASSPHRASE)
322      * 
323      * 
324      * @return True if load successfully
325      */
326     private static boolean loadSecurity() {
327         String securityName = null;
328         String securityProtocol = null;
329         String securityPassphrase = null;
330         String securityPrivProtocol = null;
331         String securityPrivPassphrase = null;
332         XmlValue value = hashConfig.get(SNMP_SECURITY);
333         @SuppressWarnings("unchecked")
334         List<XmlValue[]> list = (List<XmlValue[]>) value.getList();
335         Iterator<XmlValue[]> iterator = list.iterator();
336         while (iterator.hasNext()) {
337             securityName = null;
338             securityProtocol = null;
339             securityPassphrase = null;
340             securityPrivProtocol = null;
341             securityPrivPassphrase = null;
342             XmlValue[] subvalues = iterator.next();
343             XmlHash subHash = new XmlHash(subvalues);
344             value = subHash.get(SNMP_SECURITY_NAME);
345             if (value == null || (value.isEmpty())) {
346                 logger.warn("No Security Name found");
347                 continue;
348             }
349             securityName = value.getString();
350             value = subHash.get(SNMP_SECURITY_AUTH_PROTOCOL);
351             SecurityProtocolList secprot = null;
352             if (value != null && (!value.isEmpty())) {
353                 securityProtocol = value.getString();
354                 try {
355                     secprot = SecurityProtocolList.valueOf(securityProtocol);
356                 } catch (IllegalArgumentException e) {
357                     logger.warn("No Security Protocol found for " +
358                             securityName);
359                     continue;
360                 }
361                 value = subHash.get(SNMP_SECURITY_AUTH_PASSPHRASE);
362                 if (value == null || (value.isEmpty())) {
363                     // not allowed
364                     securityProtocol = null;
365                 } else {
366                 	securityPassphrase = value.getString();
367                 }
368             }
369             value = subHash.get(SNMP_SECURITY_PRIV_PROTOCOL);
370             PrivacyProtocolList privprot = null;
371             if (value != null && (!value.isEmpty())) {
372                 securityPrivProtocol = value.getString();
373                 try {
374                     privprot = PrivacyProtocolList
375                             .valueOf(securityPrivProtocol);
376                 } catch (IllegalArgumentException e) {
377                     logger.warn("No Security Private Protocol found for " +
378                             securityName);
379                     continue;
380                 }
381                 value = subHash.get(SNMP_SECURITY_PRIV_PASSPHRASE);
382                 if (value == null || (value.isEmpty())) {
383                     // not allowed
384                     securityPrivProtocol = null;
385                 } else {
386                 	securityPrivPassphrase = value.getString();
387                 }
388             }
389             UsmUser usm = new UsmUser(new OctetString(securityName),
390                     secprot == null? null : secprot.oid, secprot == null? null
391                             : new OctetString(securityPassphrase),
392                     privprot == null? null : privprot.oid,
393                     privprot == null? null : new OctetString(
394                             securityPrivPassphrase));
395             listUsmUser.add(usm);
396         }
397         return true;
398     }

480     /**
481      * 
482      new XmlDecl(XmlType.STRING, SNMP_TARGET_NAME), free name new
483      * XmlDecl(XmlType.STRING, SNMP_TARGET_DOMAIN), one of (Udp/Tcp)Ipv(4/6)[z]
484      * new XmlDecl(XmlType.STRING, SNMP_TARGET_ADDRESS), new
485      * XmlDecl(XmlType.INTEGER, SNMP_TARGET_TIMEOUT), new
486      * XmlDecl(XmlType.INTEGER, SNMP_TARGET_RETRIES), new
487      * XmlDecl(XmlType.BOOLEAN, SNMP_TARGET_ISV2) True => v2, else v3
488      * 
489      * new OctetString("notificationV2c"),
490      * TransportDomains.transportDomainUdpIpv4, new OctetString( new
491      * UdpAddress(toAddressV2).getValue()), 200, 1, new OctetString("notify"),
492      * new OctetString("v2c"), StorageType.permanent
493      * 
494      * new OctetString("notificationV3"),
495      * TransportDomains.transportDomainUdpIpv4, new OctetString( new
496      * UdpAddress(toAddressV3).getValue()), 200, 1, new OctetString("notify"),
497      * new OctetString("v3notify"), StorageType.permanent
498      * 
499      * @return True if successfully loaded
500      */
501     private static boolean loadTarget() {
502         String targetName;
503         String targetDomain;
504         OID oTargetDomain;
505         String targetAddress;
506         int targetTimeout;
507         int targetRetries;
508         String targetParams;
509         XmlValue value = hashConfig.get(SNMP_TARGET);
510         @SuppressWarnings("unchecked")
511         List<XmlValue[]> list = (List<XmlValue[]>) value.getList();
512         Iterator<XmlValue[]> iterator = list.iterator();
513         while (iterator.hasNext()) {
514             targetName = null;
515             targetDomain = null;
516             oTargetDomain = null;
517             targetAddress = null;
518             targetTimeout = 0;
519             targetRetries = 0;
520             targetParams = null;
521             XmlValue[] subvalues = iterator.next();
522             XmlHash subHash = new XmlHash(subvalues);
523             value = subHash.get(SNMP_TARGET_NAME);
524             if (value == null || (value.isEmpty())) {
525                 logger.warn("No Target Name found");
526                 continue;
527             }
528             targetName = value.getString();
529             value = subHash.get(SNMP_TARGET_DOMAIN);
530             if (value == null || (value.isEmpty())) {
531                 logger.warn("No Target Domain found for " + targetName);
532                 continue;
533             }
534             targetDomain = value.getString();
535             TransportDomain domain = null;
536             try {
537                 domain = TransportDomain.valueOf(targetDomain);
538                 oTargetDomain = domain.oid;
539             } catch (IllegalArgumentException e) {
540                 logger.warn("No Target Domain correctly found for " +
541                         targetName);
542                 continue;
543             }
544             value = subHash.get(SNMP_TARGET_ADDRESS);
545             if (value == null || (value.isEmpty())) {
546                 logger.warn("No Target Address found for " + targetName);
547                 continue;
548             }
549             targetAddress = value.getString();
550             TransportIpAddress address = null;
551             try {
552                 switch (domain) {
553                     case UdpIpv4:
554                     case UdpIpv6:
555                     case UdpIpv4z:
556                     case UdpIpv6z:
557                         address = new UdpAddress(targetAddress);
558                         break;
559                     case TcpIpv4:
560                     case TcpIpv6:
561                     case TcpIpv4z:
562                     case TcpIpv6z:
563                         address = new TcpAddress(targetAddress);
564                         break;
565                 }
566             } catch (IllegalArgumentException e) {
567                 logger.warn("No Correct Target Address found for " + targetName);
568                 continue;
569             }
570             logger.debug("Addr: {} {}", address.getClass(), targetAddress);
571             value = subHash.get(SNMP_TARGET_TIMEOUT);
572             if (value == null || (value.isEmpty())) {
573                 targetTimeout = 200;
574             } else {
575             	targetTimeout = value.getInteger();
576             }
577             if (targetTimeout <= 100) {
578                 targetTimeout = 100;
579             }
580             value = subHash.get(SNMP_TARGET_RETRIES);
581             if (value == null || (value.isEmpty())) {
582                 targetRetries = 1;
583             } else {
584             	targetRetries = value.getInteger();
585             }
586             if (targetRetries <= 0) targetRetries = 1;
587             value = subHash.get(SNMP_TARGET_ISV2);
588             boolean isV2 = true;
589             if (value == null || (value.isEmpty())) {
590                 isV2 = true;
591             } else {
592                 isV2 = value.getBoolean();
593             }
594             if (isV2) {
595                 hasV2 = true;
596                 targetParams = V2C;
597             } else {
598                 hasV3 = true;
599                 targetParams = V3NOTIFY;
600             }
601             TargetElement element = new TargetElement(new OctetString(
602                     targetName), oTargetDomain, new OctetString(
603                     address.getValue()), targetTimeout, targetRetries,
604                     new OctetString(NOTIFY), new OctetString(targetParams),
605                     StorageType.permanent);
606             listTargetElements.add(element);
607         }
608         return true;
609     }

611     /**
612      * Initiate the configuration from the xml file for SNMP agent
613      * 
614      * @param file
615      * @return True if OK
616      */
617     public static boolean setConfigurationFromXml(File file) {
618         Document document = null;
619         // Open config file
620         try {
621             document = new SAXReader().read(file);
622         } catch (DocumentException e) {
623             logger.error(
624                     "Unable to read the XML Config file: " +
625                             file.getAbsolutePath(), e);
626             return false;
627         }
628         if (document == null) {
629             logger.error("Unable to read the XML Config file: " +
630                     file.getAbsolutePath());
631             return false;
632         }
633         configuration = XmlUtil.read(document, configSNMP);
634         hashConfig = new XmlHash(configuration);
635         address = new String[] {
636             DEFAULTADDRESS };
637         nbThread = 4;
638         listUsmUser.clear();
639         listTargetElements.clear();
640         try {
641             // Now read the configuration
642             if (!loadConfig()) return false;
643             if (!loadSecurity()) return false;
644             if (!loadTarget()) return false;
645         } finally {
646             hashConfig.clear();
647             hashConfig = null;
648             configuration = null;
649         }
650         return true;
651     }
