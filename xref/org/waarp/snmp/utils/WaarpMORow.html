32  /**
33   * MORow implementation for GoldenGate
34   * 
35   * @author Frederic Bregier
36   * 
37   */
38  public class WaarpMORow implements MOGroup {
39      /**
40       * Row access
41       */
42      public WaarpMOScalar[] row;
43      /**
44       * Type access
45       */
46      public int[] type;
47      /**
48       * Base OID
49       */
50      public OID reference;
51      /**
52       * MIB from which this Row is issued
53       */
54      public WaarpInterfaceMib mib;
55      /**
56       * Mib Level entry identification
57       */
58      public int mibLevel;
59      /**
60       * 
61       * @param mib
62       * @param reference
63       * @param entries
64       * @param mibLevel this integer identifies this Row in the MIB
65       */
66      public WaarpMORow(WaarpInterfaceMib mib, OID reference, WaarpEntry[] entries,
67              int mibLevel) {
68          this.mib = mib;
69          this.reference = reference;
70          this.mibLevel = mibLevel;
71          row = new WaarpMOScalar[entries.length];
72          type = new int[entries.length];
73          int[] ref = this.reference.getValue();
74          int[] add = new int[2];
75          add[1] = 0;
76          for (int i = 0; i < entries.length; i ++) {
77              WaarpEntry entry = entries[i];
78              type[i] = entry.smiConstantsType;
79              add[0] = i + 1;
80              OID oid = new OID(ref, add);
81              // the value is null at the creation, meaning values have to be
82              // setup once just after
83              row[i] = WaarpMOFactory.create(oid, null, entry.smiConstantsType,
84                      entry.access, this, mibLevel, i);
85          }
86      }
87  
88      /**
89       * Set a Value in this Row
90       * @param index
91       * @param value
92       * @throws IllegalArgumentException
93       */
94      public void setValue(int index, Object value)
95              throws IllegalArgumentException {
96          if (index >= row.length)
97              throw new IllegalArgumentException("Index exceed Row size");
98          Variable var = row[index].getValue();
99          WaarpMOFactory.setVariable(var, value, type[index]);
100     }
101 
102     public void registerMOs(MOServer server, OctetString context)
103             throws DuplicateRegistrationException {
104         for (int i = 0; i < row.length; i ++) {
105             WaarpMOScalar scalar = row[i];
106             server.register(scalar, context);
107         }
108     }
109 
110     public void unregisterMOs(MOServer server, OctetString context) {
111         for (int i = 0; i < row.length; i ++) {
112             WaarpMOScalar scalar = row[i];
113             server.unregister(scalar, context);
114         }
115     }
116 }
