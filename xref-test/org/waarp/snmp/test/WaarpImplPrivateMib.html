64      /*
65       * (non-Javadoc)
66       * 
67       * @see org.waarp.snmp.GgInterfaceMib#updateServices()
68       */
69      @Override
70      public void updateServices(WaarpMOScalar scalar) {
71          // 3 groups to check
72          OID oid = scalar.getOid();
73          if (oid.startsWith(rootOIDWaarpGlobal)) {
74              // UpTime
75              if (oid.equals(rootOIDWaarpGlobalUptime)) {
76                  return;
77              }
78              ((WaarpPrivateMonitor) agent.monitor).generalValuesUpdate();
79          } else if (oid.startsWith(rootOIDWaarpDetailed)) {
80              ((WaarpPrivateMonitor) agent.monitor).detailedValuesUpdate();
81          } else if (oid.startsWith(rootOIDWaarpError)) {
82              ((WaarpPrivateMonitor) agent.monitor).errorValuesUpdate();
83          }
84      }
85  
86      /*
87       * (non-Javadoc)
88       * 
89       * @see
90       * org.waarp.snmp.GgInterfaceMib#updateServices(org.snmp4j.agent.MOScope)
91       */
92      @Override
93      public void updateServices(MOScope range) {
94          // UpTime first
95          OID low = range.getLowerBound();
96  
97          boolean okGeneral = true;
98          boolean okDetailed = true;
99          boolean okError = true;
100         if (low != null) {
101             logger.debug("low: {}:{} " + rootOIDWaarpGlobal + ":" +
102                     rootOIDWaarpDetailed + ":" + rootOIDWaarpError,
103                     low, range.isLowerIncluded());
104             if (low.size() <= rootOIDWaarp.size() &&
105                     low.startsWith(rootOIDWaarp)) {
106                 // test for global requests
107                 okGeneral = okDetailed = okError = true;
108             } else {
109                 // Test for sub requests
110                 okGeneral &= low.startsWith(rootOIDWaarpGlobal);
111                 okDetailed &= low.startsWith(rootOIDWaarpDetailed);
112                 okError &= low.startsWith(rootOIDWaarpError);
113             }
114         }
115         logger.debug("General:" + okGeneral + " Detailed:" + okDetailed +
116                 " Error:" + okError);
117         if (okGeneral) {
118             ((WaarpPrivateMonitor) agent.monitor).generalValuesUpdate();
119         }
120         if (okDetailed) {
121             ((WaarpPrivateMonitor) agent.monitor).detailedValuesUpdate();
122         }
123         if (okError) {
124             ((WaarpPrivateMonitor) agent.monitor).errorValuesUpdate();
125         }
126     }

139     /**
140      * Example of trap for All
141      * 
142      * @param message
143      * @param message2
144      * @param number
145      */
146     public void notifyInfo(String message, String message2, int number) {
147         if (!TrapLevel.All.isLevelValid(agent.trapLevel)) return;
148         logger.warn("Notify: " + NotificationElements.InfoTask + ":" + message +
149                 ":" + number);
150         agent.getNotificationOriginator().notify(
151                 new OctetString("public"),
152                 NotificationElements.InfoTask.getOID(rootOIDWaarpNotif),
153                 new VariableBinding[] {
154                         new VariableBinding(NotificationElements.InfoTask
155                                 .getOID(rootOIDWaarpNotif,
156                                         NotificationTasks.stepStatusInfo
157                                                 .getOID()), new OctetString(
158                                 message)),
159                         new VariableBinding(
160                                 NotificationElements.InfoTask
161                                         .getOID(rootOIDWaarpNotif,
162                                                 NotificationTasks.filenameInfo
163                                                         .getOID()),
164                                 new OctetString(message2)),
165                         new VariableBinding(NotificationElements.InfoTask
166                                 .getOID(rootOIDWaarpNotif,
167                                         NotificationTasks.specialIdInfo
168                                                 .getOID()), new Counter64(
169                                 number)),
170                         new VariableBinding(NotificationElements.InfoTask
171                                 .getOID(rootOIDWaarpNotif,
172                                         NotificationTasks.idRuleInfo.getOID()),
173                                 new OctetString(NotificationElements.InfoTask
174                                         .name())) });
175     }

194                                 new OctetString(message)),
195                         new VariableBinding(NotificationElements.TrapError
196                                 .getOID(rootOIDWaarpNotif, 1),
197                                 new OctetString(message2)),
198                         new VariableBinding(NotificationElements.TrapError
199                                 .getOID(rootOIDWaarpNotif, 1),
200                                 new Counter64(number)),
201                         new VariableBinding(NotificationElements.TrapError
